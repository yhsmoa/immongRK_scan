상품번호 상품바코드 상품명 발주수량 확정수량 스캔수량 입고1 입고2
function displayOrderDetail(order) { // 디버깅용 alert 추가 alert('주문 데이터 확인:\n' + JSON.stringify(order, null, 2)); const detailBody = document.getElementById('orderDetailBody'); detailBody.innerHTML = ''; order.상품정보.forEach(product => { const row = document.createElement('tr'); row.innerHTML = ` ${product.상품번호} ${product.상품바코드} ${product.상품이름} ${product.발주수량} ${product.확정수량} ${product.스캔수량 || 0} ${product.입고1 || '-'} ${product.입고2 || '-'} `; detailBody.appendChild(row); }); }
